sextime.js

1. Simple sexytime w/ auto-update

2012-10-01T03:19:00+0200

$('#test01 time').sexytime()

2. Absolute time

2012-10-01T03:19:00+0200

$('#test03 time').sexytime({ type:'absolute' })

3. Custom Callback each 60s

2012-10-01T03:19:00+0200

$('#test03 time').sexytime({ updateInterval:60, callback : function($domElm, options, momentElm, momentNow) { $domElm.html('60s interval: ' + momentElm.fromNow()); $domElm.effect('highlight', {}, 3000); }})
